NAME
       batch-disassociate-assessment-report-evidence -

DESCRIPTION

       Manager.

       See also: AWS API Documentation

          --evidence-folder-id <value>
          --evidence-ids <value>

          [--generate-cli-skeleton <value>]
          [--debug]
          [--endpoint-url <value>]
          [--no-verify-ssl]
          [--no-paginate]
          [--output <value>]
          [--query <value>]
          [--profile <value>]
          [--region <value>]
          [--version <value>]
          [--color <value>]
          [--no-sign-request]
          [--ca-bundle <value>]
          [--cli-read-timeout <value>]
          [--cli-connect-timeout <value>]
