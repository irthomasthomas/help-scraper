
Group

Subgroups:

Commands:
    create                    : Create a disk encryption set.
    delete                    : Delete a disk encryption set.
    list                      : List disk encryption sets.
    list-associated-resources : List all resources that are encrypted with this disk encryption set.
    show                      : Get information about a disk encryption set.
    update                    : Update a disk encryption set.
    wait                      : Place the CLI in a waiting state until a condition is met.
